49GameResult OctaGameLogic::makeMove(std::shared_ptr<GameCell> cell, Player player) {
50 if (!isValidMove(cell, player)) {
51 std::ostringstream oss;
52 oss << "Invalid move: ";
53 if (!cell) {
54 oss << "null cell";
55 } else if (isGameOver()) {
56 oss << "game is over";
57 } else if (player != currentPlayer_) {
58 oss << "not player's turn (current: " << static_cast<int>(currentPlayer_) + 1 << ")";
59 } else {
60 oss << "cell not owned by player " << static_cast<int>(player) + 1;
61 }
62 throw std::invalid_argument(oss.str());
63 }
64
65 // Execute the chain reaction
66 std::vector<std::shared_ptr<GameCell>> affectedCells = executeChainReaction(cell, player);
67
68 // Switch to next player
69 switchPlayer();
70
71 // Increment turn count after each move (not just after both players move)
72 // This gives us a more granular turn count for better game tracking
73 turnCount_++;
74
75 // Check for win conditions
76 auto result = checkWinConditions();
77 if (result.has_value()) {
78 gameResult_ = result;
79 gameOver_ = true;
80 return result.value();
81 }
82
83 // Return current game state
84 int p1Cells = countPlayerCells(Player::PLAYER_1);
85 int p2Cells = countPlayerCells(Player::PLAYER_2);
86
87 return GameResult("Game continues", turnCount_, p1Cells, p2Cells);

144std::vector<std::shared_ptr<GameCell>> OctaGameLogic::executeChainReaction(
145 std::shared_ptr<GameCell> startCell, Player player) {
146
147 std::vector<std::shared_ptr<GameCell>> affectedCells;
148 std::queue<std::shared_ptr<GameCell>> explosionQueue;
149
150 // Set the starting cell to the player's ownership and increment its value
151 startCell->setState(playerToCellState(player));
152 startCell->setValue(startCell->getValue() + 1);
153 affectedCells.push_back(startCell);
154
155 // Check if the starting cell should explode
156 if (shouldExplode(startCell)) {
157 explosionQueue.push(startCell);
158 }
159
160 // Process chain reactions using breadth-first search
161 while (!explosionQueue.empty()) {
162 auto currentCell = explosionQueue.front();
163 explosionQueue.pop();
164
165 // Only explode if the cell is still unstable (might have been stabilized by another explosion)
166 if (shouldExplode(currentCell)) {
167 explodeCell(currentCell, player, affectedCells);
168
169 // Check all affected neighbors for further explosions
170 for (Direction dir = Direction::N; dir <= Direction::NW;
171 dir = static_cast<Direction>(static_cast<uint8_t>(dir) + 1)) {
172
173 auto neighbor = currentCell->getNeighbor(dir);
174 if (neighbor && shouldExplode(neighbor)) {
175 // Only add to queue if not already queued (simple duplicate prevention)
176 explosionQueue.push(neighbor);
177 }
178 }
179 }
180 }
181
182 return affectedCells;
183}

193void OctaGameLogic::explodeCell(std::shared_ptr<GameCell> cell, Player player,
194 std::vector<std::shared_ptr<GameCell>>& affectedCells) {
195
196 int explosionValue = cell->getValue();
197 CellState playerState = playerToCellState(player);
198
199 // Reset the exploding cell
200 cell->setValue(0);
201 cell->setState(playerState);
202
203 // Distribute energy to all valid neighbors
204 for (Direction dir = Direction::N; dir <= Direction::NW;
205 dir = static_cast<Direction>(static_cast<uint8_t>(dir) + 1)) {
206
207 auto neighbor = cell->getNeighbor(dir);
208 if (!neighbor || neighbor->getState() == CellState::BLOCKED) {
209 continue;
210 }
211
212 // Check stopOnEnemy configuration
213 if (config_.stopOnEnemy && neighbor->getState() != CellState::NEUTRAL &&
214 neighbor->getState() != playerState) {
215 continue; // Skip enemy cells if stopOnEnemy is true
216 }
217
218 // Convert neighbor to current player and add energy
219 neighbor->setState(playerState);
220 neighbor->setValue(neighbor->getValue() + 1);
221
222 // Track affected cell
223 if (std::find(affectedCells.begin(), affectedCells.end(), neighbor) == affectedCells.end()) {
224 affectedCells.push_back(neighbor);
225 }
226 }
227}

229std::optional<GameResult> OctaGameLogic::checkWinConditions() const {
230 int p1Cells = countPlayerCells(Player::PLAYER_1);
231 int p2Cells = countPlayerCells(Player::PLAYER_2);
232
233 // Check elimination condition - but only if both players have had at least one turn
234 // This prevents immediate elimination when the game first starts
235 if (config_.winCondition == WinCondition::ELIMINATION && turnCount_ >= 2 && (p1Cells > 0 || p2Cells > 0)) {
236 if (p1Cells == 0 && p2Cells > 0) {
237 return GameResult(Player::PLAYER_2, "Player 1 eliminated", turnCount_, p1Cells, p2Cells);
238 }
239 if (p2Cells == 0 && p1Cells > 0) {
240 return GameResult(Player::PLAYER_1, "Player 2 eliminated", turnCount_, p1Cells, p2Cells);
241 }
242 }
243
244 // Check turn limit condition
245 if (config_.winCondition == WinCondition::TURN_LIMIT_MAJORITY && turnCount_ >= config_.turnLimit) {
246 if (p1Cells > p2Cells) {
247 return GameResult(Player::PLAYER_1, "Turn limit reached - Player 1 majority",
248 turnCount_, p1Cells, p2Cells);
249 }
250 if (p2Cells > p1Cells) {
251 return GameResult(Player::PLAYER_2, "Turn limit reached - Player 2 majority",
252 turnCount_, p1Cells, p2Cells);
253 }
254 return GameResult("Turn limit reached - tie", turnCount_, p1Cells, p2Cells);
255 }
256
257 return std::nullopt; // Game continues
258}

260int OctaGameLogic::countPlayerCells(Player player) const {
261 int count = 0;
262 CellState targetState = playerToCellState(player);
263
264 // Iterate through all coordinates in the map
265 // Note: This is a simplified approach. In a real implementation,
266 // we might want a more efficient way to iterate through all cells
267 // or maintain cell counts as the game progresses.
268
269 // For now, we'll use a reasonable approach assuming the map has a finite size
270 // We'll iterate through a reasonable coordinate range
271 for (int x = -10; x <= 10; ++x) {
272 for (int y = -10; y <= 10; ++y) {
273 try {
274 auto cell = gameMap_->at(Coordinate(x, y));
275 if (cell && cell->getState() == targetState) {
276 count++;
277 }
278 } catch (const std::exception&) {
279 // Coordinate not valid in map, continue
280 continue;
281 }
282 }
283 }
284
285 return count;
286}
