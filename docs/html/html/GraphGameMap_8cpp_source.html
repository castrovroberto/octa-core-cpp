39void GraphGameMap::initializeMap() {
40 // Create all cells first with NEUTRAL state and default direction
41 for (int x = -mapSize_; x <= mapSize_; ++x) {
42 for (int y = -mapSize_; y <= mapSize_; ++y) {
43 Coordinate coord(x, y);
44 if (isValidCoordinate(coord)) {
45 // Initialize cells with NEUTRAL state and N direction
46 cells_[coord] = std::make_shared<GameCell>(coord, CellState::NEUTRAL);
47 }
48 }
49 }

51 // Then link neighbors using weak_ptr references
52 for (const auto& pair : cells_) {

54 }
